  • 简单使用
a = 8 
b = ic(a)> ic| a: 8
  • 功能实现
a = 8
def half(i):
    return i/2
b = ic(half(a))> ic| half(a): 4.0
  • 列表实现
l = [1,2,3,4,5]
b = ic(l)> ic| l: [1, 2, 3, 4, 5]
  • 启用和禁用变量
ic.disable()
ic(20)
ic.enable()
ic(30)> ic| 30
  • 程序实现——毕达哥拉斯定理
import math 
n = int(input("Enter a number"))
if(n<=2):
    print("No triplets exist")
else:
    for i in range(2,n):
        for j in range(1,i):
            k = math.sqrt(i*i + j*j)
            if(k%1==0 and k<=n):
                k = int(k)
                a = ic(i,j,k)
